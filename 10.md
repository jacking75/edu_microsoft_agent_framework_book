## 10.1 구조화된 로깅

### 로깅이 중요한 이유

프로덕션 환경에서는 문제가 발생했을 때 원인을 파악할 수 있어야 한다. 좋은 로깅은 디버깅 시간을 대폭 줄여준다.

### Microsoft.Extensions.Logging 사용

```bash
dotnet add package Microsoft.Extensions.Logging
dotnet add package Microsoft.Extensions.Logging.Console
dotnet add package Serilog.Extensions.Logging
dotnet add package Serilog.Sinks.File
dotnet add package Serilog.Sinks.Console
```

### 로깅 설정

```csharp
using Microsoft.Extensions.Logging;
using Serilog;

// Serilog 설정 (파일 + 콘솔)
Log.Logger = new LoggerConfiguration()
    .MinimumLevel.Information()
    .WriteTo.Console(outputTemplate:
        "[{Timestamp:HH:mm:ss} {Level:u3}] {Message:lj}{NewLine}{Exception}")
    .WriteTo.File("logs/agent-.log",
        rollingInterval: RollingInterval.Day,
        retainedFileCountLimit: 30,
        outputTemplate: "[{Timestamp:yyyy-MM-dd HH:mm:ss} {Level:u3}] {Message:lj}{NewLine}{Exception}")
    .CreateLogger();

    builder.AddSerilog(Log.Logger));

var logger = loggerFactory.CreateLogger<Program>();

logger.LogInformation("애플리케이션 시작: {Version}", "1.0.0");
```

### 구조화된 로그 Wrapper

```csharp
public class LoggedAgent
{
    private readonly AIAgent _agent;
    private readonly ILogger _logger;
    private int _requestCount = 0;

    public LoggedAgent(AIAgent agent, ILogger logger)
    {
        _agent = agent;
        _logger = logger;
    }

    public async Task<string> RunAsync(string input, object? thread = null)
    {
        var requestId = ++_requestCount;
        var sw = System.Diagnostics.Stopwatch.StartNew();

        _logger.LogInformation(
            "Agent 요청 시작 | RequestId={RequestId} | Input={InputLength}자",
            requestId, input.Length);

        try
        {
            RunResult result;
            if (thread != null)
                result = await _agent.RunAsync(input, thread);
            else
                result = await _agent.RunAsync(input);

            sw.Stop();

            _logger.LogInformation(
                "Agent 요청 완료 | RequestId={RequestId} | Duration={Duration}ms | Output={OutputLength}자",
                requestId, sw.ElapsedMilliseconds, result.Text.Length);

            return result.Text;
        }
        catch (Exception ex)
        {
            sw.Stop();
            _logger.LogError(ex,
                "Agent 요청 실패 | RequestId={RequestId} | Duration={Duration}ms",
                requestId, sw.ElapsedMilliseconds);
            throw;
        }
    }
}
```

## 10.2 강건한 에러 처리

### 재시도 정책 (Polly 라이브러리)

```bash
dotnet add package Polly
```

```csharp
using Polly;
using Polly.Retry;

public class ResilientAgent
{
    private readonly AIAgent _agent;
    private readonly AsyncRetryPolicy _retryPolicy;
    private readonly ILogger _logger;

    public ResilientAgent(AIAgent agent, ILogger logger)
    {
        _agent = agent;
        _logger = logger;

        // 재시도 정책: 최대 3회, 지수 백오프
        _retryPolicy = Policy
            .Handle<HttpRequestException>()
            .Or<TimeoutException>()
            .WaitAndRetryAsync(
                retryCount: 3,
                sleepDurationProvider: attempt =>
                    TimeSpan.FromSeconds(Math.Pow(2, attempt)),  // 2s, 4s, 8s
                onRetry: (exception, timeSpan, attempt, context) =>
                {
                    _logger.LogWarning(
                        "요청 실패, 재시도 {Attempt}/3 (대기: {Wait}s): {Error}",
                        attempt, timeSpan.TotalSeconds, exception.Message);
                }
            );
    }

    public async Task<string> RunWithRetryAsync(string input)
    {
        return await _retryPolicy.ExecuteAsync(async () =>
        {
            var result = await _agent.RunAsync(input);
            return result.Text;
        });
    }
}
```

### 타임아웃 처리

```csharp
public async Task<string> RunWithTimeoutAsync(string input, int timeoutSeconds = 30)
{
    using var cts = new CancellationTokenSource(TimeSpan.FromSeconds(timeoutSeconds));

    try
    {
        var resultTask = _agent.RunAsync(input);
        var completedTask = await Task.WhenAny(resultTask,
            Task.Delay(Timeout.Infinite, cts.Token));

        if (completedTask == resultTask)
            return resultTask.Result.Text;
        else
            throw new TimeoutException($"{timeoutSeconds}초 내에 응답이 없다.");
    }
    catch (OperationCanceledException)
    {
        throw new TimeoutException($"{timeoutSeconds}초 내에 응답이 없다.");
    }
}
```

### 우아한 에러 응답

```csharp
public async Task<(string response, bool success)> SafeRunAsync(string input)
{
    try
    {
        var result = await RunWithRetryAsync(input);
        return (result, true);
    }
    catch (HttpRequestException ex) when (ex.Message.Contains("401"))
    {
        _logger.LogError("API 키가 유효하지 않다");
        return ("API 인증에 실패했다. 관리자에게 문의하라.", false);
    }
    catch (HttpRequestException ex) when (ex.Message.Contains("429"))
    {
        _logger.LogWarning("Rate Limit 초과");
        return ("요청이 너무 많다. 잠시 후 다시 시도하라.", false);
    }
    catch (TimeoutException)
    {
        return ("응답 시간이 초과되었다. 더 간단한 질문으로 다시 시도하라.", false);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "예상치 못한 오류");
        return ("일시적인 오류가 발생했다. 잠시 후 다시 시도하라.", false);
    }
}
```

### 입력 유효성 검사

```csharp
public static class InputValidator
{
    private static readonly string[] _blockedPatterns =
    {
        "ignore previous instructions",
        "프롬프트를 무시해",
        "시스템 지시를 변경해"
    };

    public static (bool isValid, string? error) Validate(string input)
    {
        if (string.IsNullOrWhiteSpace(input))
            return (false, "입력이 비어있다.");

        if (input.Length > 10000)
            return (false, "입력이 너무 길다 (최대 10,000자).");

        foreach (var pattern in _blockedPatterns)
        {
            if (input.Contains(pattern, StringComparison.OrdinalIgnoreCase))
                return (false, "허용되지 않는 입력이다.");
        }

        return (true, null);
    }
}
```

### Rate Limiting 구현

```csharp
public class RateLimiter
{
    private readonly SemaphoreSlim _semaphore;
    private readonly Queue<DateTime> _requestTimes = new();
    private readonly int _maxRequestsPerMinute;
    private readonly object _lock = new();

    public RateLimiter(int maxConcurrent = 5, int maxPerMinute = 60)
    {
        _semaphore = new SemaphoreSlim(maxConcurrent);
        _maxRequestsPerMinute = maxPerMinute;
    }

    public async Task<T> ExecuteAsync<T>(Func<Task<T>> action)
    {
        await _semaphore.WaitAsync();
        try
        {
            // 분당 요청 수 체크
            lock (_lock)
            {
                var now = DateTime.Now;
                var oneMinuteAgo = now.AddMinutes(-1);

                // 오래된 요청 제거
                while (_requestTimes.Count > 0 && _requestTimes.Peek() < oneMinuteAgo)
                    _requestTimes.Dequeue();

                if (_requestTimes.Count >= _maxRequestsPerMinute)
                    throw new InvalidOperationException("Rate limit 초과. 잠시 후 재시도하라.");

                _requestTimes.Enqueue(now);
            }

            return await action();
        }
        finally
        {
            _semaphore.Release();
        }
    }
}
```

## 10.4 성능 최적화

### 응답 캐싱

```csharp
public class CachedAgent
{
    private readonly AIAgent _agent;
    private readonly Dictionary<string, (string response, DateTime expires)> _cache = new();
    private readonly TimeSpan _cacheDuration;

    public CachedAgent(AIAgent agent, TimeSpan? cacheDuration = null)
    {
        _agent = agent;
        _cacheDuration = cacheDuration ?? TimeSpan.FromMinutes(10);
    }

    public async Task<string> RunAsync(string input)
    {
        var key = ComputeHash(input);

        if (_cache.TryGetValue(key, out var cached) && cached.expires > DateTime.Now)
        {
            Console.WriteLine("  [캐시 히트]");
            return cached.response;
        }

        var result = await _agent.RunAsync(input);
        _cache[key] = (result.Text, DateTime.Now.Add(_cacheDuration));
        return result.Text;
    }

    private string ComputeHash(string input)
    {
        using var sha = System.Security.Cryptography.SHA256.Create();
        var bytes = sha.ComputeHash(System.Text.Encoding.UTF8.GetBytes(input));
        return Convert.ToHexString(bytes)[..16];
    }
}
```

### 배치 처리

```csharp
public class BatchProcessor
{
    private readonly AIAgent _agent;
    private readonly int _batchSize;
    private readonly int _delayBetweenBatches;

    public BatchProcessor(AIAgent agent, int batchSize = 5,
        int delayMs = 1000)
    {
        _agent = agent;
        _batchSize = batchSize;
        _delayBetweenBatches = delayMs;
    }

    public async Task<List<string>> ProcessBatchAsync(
        IEnumerable<string> inputs)
    {
        var results = new List<string>();
        var batches = inputs.Chunk(_batchSize);

        int batchNum = 0;
        foreach (var batch in batches)
        {
            batchNum++;
            Console.WriteLine($"배치 {batchNum} 처리 중 ({batch.Length}개)...");

            var tasks = batch.Select(input => _agent.RunAsync(input));
            var batchResults = await Task.WhenAll(tasks);
            results.AddRange(batchResults.Select(r => r.Text));

            if (batchNum < batches.Count())
                await Task.Delay(_delayBetweenBatches);
        }

        return results;
    }
}
```

## 10.5 모니터링 및 알림

### 메트릭 수집

```csharp
public class AgentMetrics
{
    private long _totalRequests;
    private long _successfulRequests;
    private long _failedRequests;
    private long _totalLatencyMs;

    public void RecordSuccess(long latencyMs)
    {
        Interlocked.Increment(ref _totalRequests);
        Interlocked.Increment(ref _successfulRequests);
        Interlocked.Add(ref _totalLatencyMs, latencyMs);
    }

    public void RecordFailure()
    {
        Interlocked.Increment(ref _totalRequests);
        Interlocked.Increment(ref _failedRequests);
    }

    public void PrintReport()
    {
        var total = _totalRequests;
        var success = _successfulRequests;
        var avgLatency = total > 0 ? _totalLatencyMs / total : 0;

        Console.WriteLine($"""
📊 Agent 성능 보고서:
  총 요청: {total:N0}
  성공: {success:N0} ({(total > 0 ? 100.0 * success / total : 0):F1}%)
  실패: {_failedRequests:N0}
  평균 응답 시간: {avgLatency}ms
""");
    }
}
```

## 10.7 완성된 ProductionAgent.cs

```csharp
using DotNetEnv;
using Microsoft.Extensions.Logging;
using AgentFrameworkBook.Shared;

Env.Load();

// 로깅 설정
using var loggerFactory = LoggerFactory.Create(builder =>
    builder.AddConsole().SetMinimumLevel(LogLevel.Information));
var logger = loggerFactory.CreateLogger<Program>();

logger.LogInformation("ProductionAgent 시작");

// Agent 생성
var baseAgent = AIAgentBuilder.FromEnvironment()
    .Build("ProductionAgent", "프로덕션 레벨의 안정적인 어시스턴트다.");

// 프로덕션 래퍼 적용
var metrics = new AgentMetrics();
var rateLimiter = new RateLimiter(maxConcurrent: 3, maxPerMinute: 30);
var resilientAgent = new ResilientAgent(baseAgent, logger);

Console.WriteLine("✅ ProductionAgent 준비 완료");
Console.WriteLine("종료: 'quit' | 메트릭: 'stats'\n");

while (true)
{
    Console.Write("💬 입력: ");
    var input = Console.ReadLine()?.Trim();

    if (string.IsNullOrWhiteSpace(input)) continue;
    if (input == "quit") break;
    if (input == "stats") { metrics.PrintReport(); continue; }

    var (isValid, error) = InputValidator.Validate(input);
    if (!isValid) { Console.WriteLine($"⚠️ {error}\n"); continue; }

    var sw = System.Diagnostics.Stopwatch.StartNew();
    try
    {
        var response = await rateLimiter.ExecuteAsync(async () =>
            await resilientAgent.RunWithRetryAsync(input));

        sw.Stop();
        metrics.RecordSuccess(sw.ElapsedMilliseconds);

        Console.WriteLine($"\n🤖 Agent: {response}\n");
    }
    catch (Exception ex)
    {
        sw.Stop();
        metrics.RecordFailure();
        logger.LogError(ex, "요청 실패");
        Console.WriteLine($"\n❌ 오류: {ex.Message}\n");
    }
}

metrics.PrintReport();
logger.LogInformation("ProductionAgent 종료");
```
